class Sever():
    def __init__(self,m,n,d):
        self.m=m # number of worker
        self.n=n # number of sample
        self.d=d # number of features
        self.w=numpy.zeros((d,1))
        self.g=numpy.zeros((d,1))
        self.v=numpy.zeros((d,1))
        self.loss=0
        self.obj=0
    
    def broadcast(self):
        return self.w
    
    '''def gradient(self,lam):
        self.g=self.g/self.n+lam*self.w'''
    
    def aggregate(self,direction,losses):
        self.direction=numpy.zeros((self.d,1))
        self.loss=0
        for i in range(self.m):
            #get the direction
            self.direction+=direction[i]
            self.loss+=losses[i]
        # the gradient
        self.direction=self.direction/self.m
    
    def gd(self,alpha):
        #self.v*=beta
        #self.v+=self.g
        self.w-=alpha*self.direction
        
    def objective(self,lam):
        reg=lam/2*numpy.sum(self.w*self.w)
        self.obj=self.loss/self.n+reg
        return self.obj

class Worker:
    def __init__(self,x,y):
        self.x=x
        self.y=y
        self.s=x.shape[0] # number of sample
        self.d=x.shape[1] # number of features
        #self.r=r # running time
        self.w=numpy.zeros((self.d,1))
        self.old_w=numpy.zeros((self.d,1))
        
        #this is for federate averaging
        self.w=numpy.zeros((self.d,1))
        self.g=numpy.zeros((self.d,1))
        self.v=numpy.zeros((self.d,1))
        
    def set_param(self,w):
        #self.w=copy.deepcopy(w)
        self.w=w
        self.old_w=copy.deepcopy(w)
        
    '''def loss(self):
        yx=numpy.multiply(self.y,self.x)
        yxw=numpy.dot(yx,self.w)
        vec1=numpy.exp(-yxw)
        vec2=numpy.log(1+vec1)
        return numpy.sum(vec2)'''
    
    def loss(self,x,y):
        yx=numpy.multiply(y,x)
        yxw=numpy.dot(yx,self.w)
        vec1=numpy.exp(-yxw)
        vec2=numpy.log(1+vec1)
        return numpy.sum(vec2)
    
    def gradient(self,lam,x,y):
        yx=numpy.multiply(y,x)
        yxw=numpy.dot(yx,self.w)
        vec1=numpy.exp(yxw)
        vec2=numpy.divide(yx,1+vec1)
        # this is the mean of the gradient
        g=-numpy.mean(vec2,axis=0).reshape(self.d,1)
        # the result of the self iteration
        g=g+lam*self.w
        return g
        #return g
    
    
    def gd(self,alpha,lam):
        #self.v*=beta
        #self.v+=self.g
        l=self.loss(self.x,self.y)
        self.g=self.gradient(lam,self.x,self.y)
        self.w-=alpha*self.g
        return l
    
    def sgd(self,alpha,lam):
        loss=0
        
        #count=[]
        self.g=numpy.zeros((self.d,1))
        for i in range(self.s):
            #print(i)
            xi = self.x[i, :] # 1-by-d matrix
            yi = float(self.y[i, :]) # scalar
            l = self.loss(xi,yi)
            # loss
            loss+=l
            grad=self.gradient(lam,xi,yi)
            self.w-=alpha*grad
        return loss#,count
            
    
    def dirct(self):
        return self.old_w-self.w
